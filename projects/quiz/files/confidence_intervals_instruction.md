```python
import numpy as np
from scipy.stats import norm

def confidence_interval(p, n, confidence_level=0.95):
    z_score = norm.ppf((1 + confidence_level) / 2)
    se = np.sqrt(p * (1 - p) / n)
    moe = z_score * se
    return (p - moe, p + moe)
```
